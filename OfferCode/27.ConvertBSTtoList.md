> 题目描述

> 输入一棵二叉搜索树，将该二叉搜索树转换成一个排序的双向链表。要求不能创建任何新的结点，只能调整树中结点指针的指向。

**二叉搜索树的特性：中序遍历是递增序列，下面这个方法需要额外O(n)空间**

```c++
class Solution {
public:
    TreeNode* Convert(TreeNode* root)
    {
        if(root == NULL) return root;
        v.clear();
		dfs(root);
        for(int i = 0; i < v.size(); i++){
            if(i == 0) v[i]->left = NULL;
            else v[i]->left = v[i-1];
            if(i == v.size()-1) v[i]->right = NULL;
            else v[i]->right = v[i+1];
        }
        return v[0];
    }
private:
    vector<TreeNode*> v;
    void dfs(TreeNode* root){
		if(root == NULL) return;
        dfs(root->left);
        v.push_back(root);
        dfs(root->right);
    }
};
```

<br/>
**convert方法返回以root为根的BST转换成的双向链表的头指针，那么只需要找到左子树转换后链表的尾指针，和根结点链接起来，然后更新当前链表的尾结点，然后去遍历转换右子树，把根结点和右子树中最小结点链接起来。**

```c++
TreeNode *lastNodeInList = NULL;
TreeNode* Convert(TreeNode* root)
{
    if(root == NULL) return root;
    if(root->left == NULL && root->right == NULL) {
        lastNodeInList = root;
        return root;
    }

    TreeNode *left = Convert(root->left);
    if(left != NULL){
        lastNodeInList->right = root;
        root->left = lastNodeInList;
    }
    lastNodeInList = root;
    TreeNode *right = Convert(root->right);
    if(right != NULL){
        root->right = right;
        right->left = root;
    }
    return left == NULL ? root : left;
}
```

**其实链接根结点和右子树头指针的代码可以省略的，因为更新尾指针之后，遍历右子树也是先遍历右子树的左子树**

/*

class Solution {

    TreeNode* Convert(TreeNode* root)
    {
        dfs(root);
        return getHead(lastNodeInList);
    }
private:
    TreeNode *lastNodeInList = NULL;
    void dfs(TreeNode *root){
        if(root == NULL) return;
        dfs(root->left);
        root->left = lastNodeInList;
        if(lastNodeInList != NULL) lastNodeInList->right = root;
        lastNodeInList = root;
        dfs(root->right);
    }
    TreeNode* getHead(TreeNode *tail){
        while(tail != NULL && tail->left != NULL){
            tail = tail->left;
        }
        return tail;
    }
};
